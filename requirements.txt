# requirements.txt — Versões pinadas para o projeto FOTECTA
#
# Este arquivo contém as dependências bloqueadas do projeto e instruções
# rápidas para criar o ambiente, instalar as versões exatas e publicar
# um release no GitHub (macOS e Windows).
# NÃO inclua tokens ou segredos neste arquivo.
#
# macOS (Terminal) — criar/ativar venv e instalar tudo
# 1. Navegue até a pasta do projeto:
#    cd /path/to/projeto-integrador-DRP01
# 2. Crie e ative o ambiente virtual:
#    python3 -m venv venv
#    . venv/bin/activate
# 3. Atualize o pip e instale as dependências:
#    python -m pip install --upgrade pip
#    pip install -r requirements.txt
#
# Windows (PowerShell) — criar/ativar venv e instalar tudo
# 1. Navegue até a pasta do projeto:
#    cd C:\path\to\projeto-integrador-DRP01
# 2. Crie e ative o ambiente virtual (PowerShell):
#    python -m venv venv
#    Set-ExecutionPolicy -Scope Process -ExecutionPolicy Bypass
#    . .\venv\Scripts\Activate.ps1
#  (No CMD use: venv\Scripts\activate.bat)
# 3. Atualize o pip e instale as dependências:
#    python -m pip install --upgrade pip
#    pip install -r requirements.txt
#
# Instalar um pacote específico (qualquer SO):
#    pip install <pacote>==<versão>
# Exemplo:
#    pip install Flask==3.1.3
#
# Verificar versões instaladas:
# macOS/Linux:
#    pip show Flask
#    pip freeze | grep -i Flask
# Windows (PowerShell):
#    pip show Flask
#    pip freeze | findstr /I Flask
#
# Gerar/atualizar requirements.txt (após instalar ou atualizar pacotes):
#    . venv/bin/activate       # macOS/Linux
#    .\venv\Scripts\Activate.ps1  # PowerShell
#    pip freeze > requirements.txt
#    git add requirements.txt
#    git commit -m "Update requirements"
#    git push
#
# Criar release no GitHub (recomendado: gh CLI)
# macOS (Homebrew) / Windows (winget):
#    # macOS
#    brew install gh
#    # Windows (PowerShell admin)
#    winget install --id GitHub.cli
#
# Autenticar e criar release com gh:
#    gh auth login
#    gh repo set-default <owner>/<repo>   # ex: victorxmdev/projeto-integrador-DRP01
#    gh release create v1.0.0 -t "v1.0.0" -n "Release criada; inclui requirements.txt"
#
# Alternativa via API (curl) usando um PAT — NÃO cole o token em arquivos públicos:
#    export GITHUB_TOKEN="SEU_PAT_AQUI"   # macOS/Linux
#    setx GITHUB_TOKEN "SEU_PAT_AQUI"    # Windows (ou use set in cmd)
#    curl -X POST \
#      -H "Authorization: token $GITHUB_TOKEN" \
#      -H "Accept: application/vnd.github+json" \
#      https://api.github.com/repos/<owner>/<repo>/releases \
#      -d '{"tag_name":"v1.0.0","name":"v1.0.0","body":"Release criada; inclui requirements.txt","draft":false,"prerelease":false}'
#
# Verificar release no navegador:
#    open "https://github.com/<owner>/<repo>/releases/tag/v1.0.0"   # macOS
#    start "https://github.com/<owner>/<repo>/releases/tag/v1.0.0"  # Windows
#
# Dicas de segurança:
#  - Nunca cole tokens no chat ou no código versionado.
#  - Use o gh auth login para autenticação interativa.
#
